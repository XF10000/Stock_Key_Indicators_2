```python
# 验证器示例代码
def validate_stock_code(code: str) -> bool:
    """验证股票代码格式"""
    return bool(re.match(r'^\d{6}$', code))

def validate_report_date(date_str: str) -> bool:
    """验证报告日期格式"""
    try:
        date = datetime.strptime(date_str, '%Y-%m-%d')
        return date.month in [3, 6, 9, 12] and date.day in [31, 30]
    except ValueError:
        return False

def validate_numeric_range(value: float, min_val: float = None, max_val: float = None) -> bool:
    """验证数值范围"""
    if value is None or pd.isna(value):
        return True  # 空值不验证范围
    if min_val is not None and value < min_val:
        return False
    if max_val is not None and value > max_val:
        return False
    return True
```

#### 检查方法

```python
def check_accounting_equation(balance_df: pd.DataFrame, tolerance: float = 0.001) -> dict:
    """检查会计恒等式"""
    results = {
        'passed': 0,
        'failed': 0,
        'errors': []
    }
    
    for idx, row in balance_df.iterrows():
        assets = row.get('total_assets', 0) or 0
        liabilities = row.get('total_liabilities', 0) or 0
        equity = row.get('total_owners_equity', 0) or 0
        
        expected = liabilities + equity
        if assets == 0:
            continue  # 跳过空数据
        
        error_rate = abs(assets - expected) / assets
        
        if error_rate > tolerance:
            results['failed'] += 1
            results['errors'].append({
                'report_date': row['report_date'],
                'assets': assets,
                'expected': expected,
                'error_rate': error_rate
            })
        else:
            results['passed'] += 1
    
    return results

def check_report_continuity(dates: list, years: int = 10) -> dict:
    """检查报告期连续性"""
    expected_count = years * 4
    actual_count = len(dates)
    
    return {
        'expected': expected_count,
        'actual': actual_count,
        'complete': actual_count >= expected_count,
        'missing_periods': expected_count - actual_count
    }
```

```python
class DataExceptionHandler:
    """异常数据处理策略"""
    
    def handle_missing_field(self, field_name: str, context: dict) -> None:
        """处理字段缺失"""
        # 输出 "-" 标记
        # 记录WARNING日志
        logging.warning(f"字段缺失: {field_name}, 股票: {context.get('stock_code')}")
    
    def handle_invalid_value(self, field_name: str, value: float, reason: str) -> float:
        """处理无效数值"""
        if field_name == 'effective_tax_rate':
            return None  # 税率无效则返回None
        return value  # 其他字段保留原值
    
    def handle_ttm_calculation_failure(self, stock_code: str, reason: str) -> None:
        """处理TTM计算失败"""
        logging.info(f"TTM计算跳过: {stock_code}, 原因: {reason}")
        # TTM列输出 "-"
    
    def handle_validation_failure(self, stock_code: str, errors: list) -> bool:
        """处理校验失败"""
        # 记录详细错误
        for error in errors:
            logging.error(f"校验失败: {stock_code}, {error}")
        
        # 根据错误严重程度决定是否继续
        critical_errors = [e for e in errors if e.get('level') == 'CRITICAL']
        return len(critical_errors) == 0  # 无严重错误则继续
```

```python
def process_with_memory_monitor(stock_code: str):
    """带内存监控的处理"""
    process = psutil.Process()
    start_memory = process.memory_info().rss / 1024 / 1024  # MB
    
    try:
        # 读取数据
        data = read_financial_data(stock_code)
```

#### 内存监控

```python
class MemoryMonitor:
    """内存监控器"""
    
    def __init__(self, warning_threshold: float = 0.7, critical_threshold: float = 0.85):
        self.warning_threshold = warning_threshold
        self.critical_threshold = critical_threshold
        self.process = psutil.Process()
    
    def get_memory_usage(self) -> float:
        """获取内存使用率"""
        return self.process.memory_info().rss / psutil.virtual_memory().total
    
    def check_memory(self) -> tuple[str, float]:
        """检查内存状态"""
        usage = self.get_memory_usage()
        
        if usage > self.critical_threshold:
            return 'CRITICAL', usage
        elif usage > self.warning_threshold:
            return 'WARNING', usage
        else:
            return 'OK', usage
    
    def log_memory_status(self) -> None:
        """记录内存状态"""
        status, usage = self.check_memory()
        
        if status == 'CRITICAL':
            logging.critical(f"内存使用率过高: {usage:.1%}")
        elif status == 'WARNING':
            logging.warning(f"内存使用率较高: {usage:.1%}")
        else:
            logging.debug(f"内存使用率正常: {usage:.1%}")
```

### 3.3 内存阈值和降级策略

| 阈值级别 | 内存使用率 | 处理方式 |
|----------|------------|----------|
| 正常 | <70% | 正常处理 |
| 警告 | 70%-85% | 记录日志，继续处理 |
| 严重 | 85%-95% | 暂停新任务，强制GC，减小批次大小 |
| 危险 | >95% | 立即停止，保存进度，等待内存释放 |

```python
def handle_memory_pressure(level: str, usage: float):
    """处理内存压力"""
    if level == 'WARNING':
        logging.warning(f"内存使用率 {usage:.1%}，建议监控")
        # 继续处理，但减小批次
        return True
    
    elif level == 'CRITICAL':
        logging.error(f"内存使用率 {usage:.1%}，暂停处理")
        # 强制GC
        gc.collect()
        # 等待内存释放
        time.sleep(1)
        # 检查是否恢复
        new_usage = get_memory_usage()
        return new_usage < 0.85
    
    elif level == 'DANGER':
        logging.critical(f"内存使用率 {usage:.1%}，立即停止")
        # 保存进度
        save_progress()
        # 停止处理
        return False
```

```python
    # 输出结果
    all_passed = all(c[1] for c in checks)
    for name, passed, error in checks:
        status = "✅" if passed else "❌"
        print(f"{status} {name}")
        if error:
            print(f"   错误: {error}")
    
    return all_passed
```

| 亏损公司 | 连续亏损 | 负值正确处理 |
| 数据缺失严重 | 字段大量缺失 | 标记缺失，不崩溃 |
