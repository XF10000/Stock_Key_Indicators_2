```python
# analysis/restructure/ttm.py

from typing import Optional

import pandas as pd


class TTCalculator:
    """TTM（滚动12个月）计算器
    
    计算利润表和现金流量表的TTM数据。
    
    TTM计算公式：
        TTM = 今年Q3累计 - 去年Q3累计 + 去年Q4累计
    
    Example:
        >>> calculator = TTCalculator()
        >>> ttm_df = calculator.calculate(income_df)
        >>> print(ttm_df.columns[-1])
        '2025Q3-TTM'
    """
    
    def calculate(
        self, 
        df: pd.DataFrame,
        date_column: str = "report_date"
    ) -> pd.DataFrame:
        """计算TTM数据
        
        为DataFrame添加TTM列，仅计算最新报告期的TTM。
        
        Args:
            df: 原始数据DataFrame（包含累计值）
            date_column: 日期列名
        
        Returns:
            添加TTM列后的DataFrame
        
        Note:
            - 数据缺失时TTM值为None
            - 新股/次新股TTM值为None
            - 仅计算最新报告期，历史数据使用年报
        
        Example:
            >>> income_with_ttm = calculator.calculate(income_df)
            >>> print(income_with_ttm["2025Q3-TTM"]["operating_revenue"])
            1234567890.0
        """
        ...
    
    def calculate_single_value(
        self,
        current_q3: Optional[float],
        last_year_q3: Optional[float],
        last_year_q4: Optional[float]
    ) -> Optional[float]:
        """计算单个字段的TTM值
        
        Args:
            current_q3: 今年Q3累计值
            last_year_q3: 去年Q3累计值
            last_year_q4: 去年Q4累计值（年报）
        
        Returns:
            TTM值，数据缺失时返回None
        
        Example:
            >>> ttm = calculator.calculate_single_value(100, 60, 80)
            >>> print(ttm)
            120.0  # 100 - 60 + 80 = 120
        """
        ...
    
    def is_ttm_valid(
        self,
        current_q3: Optional[float],
        last_year_q3: Optional[float],
        last_year_q4: Optional[float]
    ) -> bool:
        """判断TTM计算是否有效
        
        检查所需数据是否完整。
        
        Args:
            current_q3: 今年Q3累计值
            last_year_q3: 去年Q3累计值
            last_year_q4: 去年Q4累计值
        
        Returns:
            True表示数据完整，可计算TTM
        
        Example:
            >>> calculator.is_ttm_valid(100, None, 80)
            False  # 缺少去年Q3数据
        """
        ...
```

```python
# analysis/restructure/rules.py

from dataclasses import dataclass
from pathlib import Path
from typing import Any, Optional

import yaml


@dataclass
class RestructureRule:
    """重构规则定义
    
    定义单个重构项的计算规则。
    
    Attributes:
        name: 项目名称（中文）
        field: 输出字段名
        formula: 计算公式（支持加减法表达式）
        category: 所属分类
        is_subtotal: 是否为小计项
        source_fields: 源字段列表
    """
    name: str
    field: str
    formula: str
    category: str
    is_subtotal: bool = False
    source_fields: list[str] = None


@dataclass
class RestructureConfig:
    """重构配置
    
    存储完整的重构规则配置。
    
    Attributes:
        balance_sheet_rules: 资产负债表规则
        income_statement_rules: 利润表规则
        cash_flow_rules: 现金流量表规则
        equity_cost_rate: 股权资本成本率（默认8%）
    """
    balance_sheet_rules: list[RestructureRule]
    income_statement_rules: list[RestructureRule]
    cash_flow_rules: list[RestructureRule]
    equity_cost_rate: float = 0.08


class RuleEngine:
    """规则引擎
    
    加载YAML配置文件，解析重构规则，执行计算。
    
    Example:
        >>> engine = RuleEngine.from_yaml("rules.yaml")
        >>> result = engine.calculate_field("金融资产合计", data_df)
        >>> print(result)
        1234567890.0
    """
    
    @classmethod
    def from_yaml(cls, config_path: str | Path) -> "RuleEngine":
        """从YAML文件加载规则
        
        Args:
            config_path: YAML配置文件路径
        
        Returns:
            RuleEngine实例
        
        Raises:
            FileNotFoundError: 配置文件不存在
            YAMLError: YAML格式错误
        
        Example:
            >>> engine = RuleEngine.from_yaml("analysis/restructure/rules.yaml")
        """
        ...
    
    def __init__(self, config: RestructureConfig) -> None:
        """初始化规则引擎
        
        Args:
            config: 重构配置对象
        """
        ...
    
    def calculate_field(
        self,
        rule: RestructureRule,
        data: pd.DataFrame,
        row_index: int = 0
    ) -> Optional[float]:
        """计算单个字段值
        
        根据规则公式计算字段值。
        
        Args:
            rule: 重构规则
            data: 数据DataFrame
            row_index: 行索引（默认0，即最新报告期）
        
        Returns:
            计算结果，数据缺失时返回None
        
        Example:
            >>> value = engine.calculate_field(rule, df)
        """
        ...
    
    def get_rule_by_name(
        self, 
        name: str, 
        table_type: str
    ) -> Optional[RestructureRule]:
        """按名称获取规则
        
        Args:
            name: 项目名称
            table_type: 表类型（balance_sheet/income_statement/cash_flow）
        
        Returns:
            规则对象，未找到返回None
        """
        ...
    
    def get_rules_by_category(
        self,
        category: str,
        table_type: str
    ) -> list[RestructureRule]:
        """按分类获取规则列表
        
        Args:
            category: 分类名称
            table_type: 表类型
        
        Returns:
            规则列表
        """
```

```python
# analysis/restructure/income_statement.py

from dataclasses import dataclass
from typing import Optional

import pandas as pd

from .rules import RuleEngine
from .ttm import TTCalculator


class IncomeStatementRestructurer:
    """利润表重构器
    
    将传统利润表重构为股权价值增加表。
    
    重构结构：
        - 营业收入 → 毛利 → 息税前经营利润
        - 投资收益 → 息税前金融资产收益
```

```python
    """主入口函数
    
    Returns:
        退出码（0成功，非0失败）
    """
    args = parse_args()
    setup_logging()
    logger = logging.getLogger(__name__)
    
    # 初始化引擎
    engine = RestructureEngine(
        db_path=args.db_path,
        rules_path=args.rules_path,
        equity_cost_rate=args.equity_cost_rate
    )
    
    # 确定股票列表
    if args.code:
        stock_codes = [args.code]
    elif args.codes == "all":
        stock_codes = engine.repository.get_stock_list()
    else:
        stock_codes = [c.strip() for c in args.codes.split(",")]
    
    # 断点续传处理
    if args.start_from:
        try:
            start_index = stock_codes.index(args.start_from)
            stock_codes = stock_codes[start_index:]
            logger.info(f"从 {args.start_from} 继续，剩余 {len(stock_codes)} 家公司")
        except ValueError:
            logger.error(f"起始代码 {args.start_from} 不在列表中")
            return 1
    
    # 执行重构
    success_count = 0
    failed_count = 0
    
    with tqdm(stock_codes, desc="处理进度") as pbar:
        for stock_code in pbar:
            try:
                result = engine.restructure(
                    stock_code=stock_code,
                    years=args.years,
                    output_dir=args.output_dir
                )
                success_count += 1
                logger.info(f"{stock_code} 处理完成: {result.output_path}")
            except Exception as e:
                failed_count += 1
                logger.error(f"{stock_code} 处理失败: {e}")
            
            pbar.set_postfix({
                "成功": success_count,
                "失败": failed_count
            })
    
    # 输出摘要
    logger.info(f"处理完成: 成功 {success_count}, 失败 {failed_count}")
    
    return 0 if failed_count == 0 else 1


if __name__ == "__main__":
    sys.exit(main())
```

```python
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
import pandas as pd
from datetime import date

def create_restructure_excel(
    stock_code: str,
    balance_sheet: pd.DataFrame,
    income_statement: pd.DataFrame,
    cash_flow: pd.DataFrame,
    output_dir: str = "output/restructure"
) -> Path:
    """生成重构报表Excel文件"""
    
    # 创建工作簿
    wb = Workbook()
    
    # 样式定义
    header_font = Font(name="微软雅黑", size=11, bold=True)
    header_fill = PatternFill(start_color="D9E1F2", end_color="D9E1F2", fill_type="solid")
    header_alignment = Alignment(horizontal="center", vertical="center")
    data_font = Font(name="微软雅黑", size=10)
    subtotal_font = Font(name="微软雅黑", size=10, bold=True)
    subtotal_fill = PatternFill(start_color="F2F2F2", end_color="F2F2F2", fill_type="solid")
    number_alignment = Alignment(horizontal="right", vertical="center")
    border = Border(
        left=Side(style="thin"),
        right=Side(style="thin"),
        top=Side(style="thin"),
        bottom=Side(style="thin")
    )
    
    # 写入三个Sheet
    for sheet_name, df in [
        ("资产资本表", balance_sheet),
        ("股权价值增加表", income_statement),
        ("现金流量分析表", cash_flow)
    ]:
        ws = wb.create_sheet(title=sheet_name)
        
        # 写入数据
        for r_idx, row in enumerate(dataframe_to_rows(df, index=False, header=True), 1):
            for c_idx, value in enumerate(row, 1):
                cell = ws.cell(row=r_idx, column=c_idx, value=value)
                cell.border = border
                
                # 应用样式
                if r_idx == 1:  # 表头行
                    cell.font = header_font
                    cell.fill = header_fill
                    cell.alignment = header_alignment
                else:
                    cell.font = data_font
                    if c_idx == 1:  # 项目列
                        cell.alignment = Alignment(horizontal="left", vertical="center")
                        # 检查是否为小计行
                        if "合计" in str(value) or "小计" in str(value):
                            cell.font = subtotal_font
                            for c in range(1, len(row) + 1):
                                ws.cell(row=r_idx, column=c).fill = subtotal_fill
                    else:  # 数值列
                        cell.alignment = number_alignment
                        if isinstance(value, float):
                            cell.number_format = "#,##0.00"
        
        # 设置列宽
        ws.column_dimensions['A'].width = 30
        for col in range(2, len(df.columns) + 2):
            ws.column_dimensions[chr(64 + col)].width = 15
    
    # 删除默认Sheet
    if "Sheet" in wb.sheetnames:
        del wb["Sheet"]
    
    # 保存文件
    output_path = Path(output_dir) / f"{stock_code}_重构报表_{date.today()}.xlsx"
    output_path.parent.mkdir(parents=True, exist_ok=True)
    wb.save(output_path)
    
    return output_path
```
