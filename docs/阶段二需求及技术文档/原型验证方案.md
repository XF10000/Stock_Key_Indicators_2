```python
"""
原型步骤2: 验证数据读取

from sqlalchemy import create_engine
from sqlalchemy.orm import Session
from models import BalanceSheet, IncomeStatement, CashFlowStatement
import pandas as pd

def read_financial_data(stock_code: str, years: int = 5):
    """读取单家公司财务数据"""
    engine = create_engine("sqlite:///database.sqlite")
    
    with Session(engine) as session:
        # 读取资产负债表
        balance_query = session.query(BalanceSheet).filter(
            BalanceSheet.stock_code == stock_code
        ).order_by(BalanceSheet.report_date.desc()).limit(years * 4)
        
        balance_df = pd.read_sql(
            balance_query.statement, 
            engine
        )
        
        # 读取利润表
        income_query = session.query(IncomeStatement).filter(
            IncomeStatement.stock_code == stock_code
        ).order_by(IncomeStatement.report_date.desc()).limit(years * 4)
        
        income_df = pd.read_sql(
            income_query.statement,
            engine
        )
        
        # 读取现金流量表
        cashflow_query = session.query(CashFlowStatement).filter(
            CashFlowStatement.stock_code == stock_code
        ).order_by(CashFlowStatement.report_date.desc()).limit(years * 4)
        
        cashflow_df = pd.read_sql(
            cashflow_query.statement,
            engine
        )
    
    return balance_df, income_df, cashflow_df
```

```python
"""
原型步骤3: 验证TTM计算
"""
import pandas as pd
from datetime import date

def calculate_ttm(df: pd.DataFrame, value_column: str) -> pd.DataFrame:
    """
    计算TTM值
    
    TTM = 今年Q3累计 - 去年Q3累计 + 去年Q4累计
    """
    df = df.copy()
    df['report_date'] = pd.to_datetime(df['report_date'])
    df = df.sort_values('report_date', ascending=False)
    
    # 找到最新Q3
    latest = df.iloc[0]
    latest_date = latest['report_date']
    
    # 如果不是Q3，返回None
    if latest_date.month != 9:
        print(f"最新报告期不是Q3: {latest_date}")
        return df
    
    # 找到去年Q3和Q4
    current_year = latest_date.year
    last_year_q3_date = date(current_year - 1, 9, 30)
    last_year_q4_date = date(current_year - 1, 12, 31)
    
    # 获取各期数据
    current_q3 = df[df['report_date'] == pd.Timestamp(latest_date)]
    last_year_q3 = df[df['report_date'] == pd.Timestamp(last_year_q3_date)]
    last_year_q4 = df[df['report_date'] == pd.Timestamp(last_year_q4_date)]
    
    if len(current_q3) == 0 or len(last_year_q3) == 0 or len(last_year_q4) == 0:
        print("数据不完整，无法计算TTM")
        return df
    
    # 计算TTM
    current_value = current_q3.iloc[0][value_column]
    last_q3_value = last_year_q3.iloc[0][value_column]
    last_q4_value = last_year_q4.iloc[0][value_column]
    
    if pd.isna(current_value) or pd.isna(last_q3_value) or pd.isna(last_q4_value):
        ttm_value = None
    else:
        ttm_value = current_value - last_q3_value + last_q4_value
    
    # 添加TTM列
    df['TTM'] = None
    df.iloc[0, df.columns.get_loc('TTM')] = ttm_value
    
    print(f"TTM计算结果:")
    print(f"  今年Q3累计: {current_value:,.0f}")
    print(f"  去年Q3累计: {last_q3_value:,.0f}")
    print(f"  去年Q4累计: {last_q4_value:,.0f}")
    print(f"  TTM: {ttm_value:,.0f}" if ttm_value else "  TTM: None")
    
    return df

if __name__ == "__main__":
    # 使用步骤2的数据
    from prototype_step2_read_data import read_financial_data
    
    balance, income, cashflow = read_financial_data("000333", years=5)
    
    # 计算营业收入的TTM
```

```python
"""
原型步骤4: 验证资产负债表重构
"""
import pandas as pd

def restructure_balance_sheet(balance_df: pd.DataFrame) -> pd.DataFrame:
    """
    重构资产负债表为资产资本表
    
    结构：
    - 金融资产合计
    - 长期股权投资
    - 经营资产合计
    - 有息债务合计
    - 所有者权益合计
    """
    # 取最新报告期数据
    latest = balance_df.iloc[0]
    
    # 计算金融资产
    financial_assets = {
        '货币资金': latest.get('monetary_capital', 0) or 0,
        '交易性金融资产': latest.get('trading_financial_assets', 0) or 0,
        '衍生金融资产': latest.get('derivative_financial_assets', 0) or 0,
        # ... 更多金融资产
    }
    
    # 计算金融资产合计
    total_financial = sum(financial_assets.values())
    
    # 计算经营资产
    operating_assets = {
        '应收票据': latest.get('notes_receivable', 0) or 0,
        '应收账款': latest.get('accounts_receivable', 0) or 0,
        '存货': latest.get('inventories', 0) or 0,
        # ... 更多经营资产
    }
    
    # 计算经营负债
    operating_liabilities = {
        '应付票据': latest.get('notes_payable', 0) or 0,
        '应付账款': latest.get('accounts_payable', 0) or 0,
        # ... 更多经营负债
    }
    
    # 周转性经营投入 = 经营资产 - 经营负债
    working_capital = sum(operating_assets.values()) - sum(operating_liabilities.values())
    
    # 计算有息债务
    interest_debt = {
        '短期借款': latest.get('short_term_borrowings', 0) or 0,
        '长期借款': latest.get('long_term_borrowings', 0) or 0,
        '应付债券': latest.get('bonds_payable', 0) or 0,
    }
    total_debt = sum(interest_debt.values())
    
    # 构建重构结果
    result = {
        '项目': [],
        '金额': [],
    }
    
    # 添加金融资产项
    for name, value in financial_assets.items():
        if value != 0:
            result['项目'].append(name)
            result['金额'].append(value)
    
    result['项目'].append('金融资产合计')
    result['金额'].append(total_financial)
    
    # 添加经营资产项
    result['项目'].append('周转性经营投入合计')
    result['金额'].append(working_capital)
    
    # 添加有息债务项
    for name, value in interest_debt.items():
        if value != 0:
            result['项目'].append(name)
            result['金额'].append(value)
    
    result['项目'].append('有息债务合计')
    result['金额'].append(total_debt)
    
    # 添加所有者权益
    result['项目'].append('所有者权益合计')
    result['金额'].append(latest.get('total_owners_equity', 0) or 0)
    
    return pd.DataFrame(result)

def validate_restructure(original: pd.DataFrame, restructured: pd.DataFrame) -> bool:
    """验证重构结果"""
    latest = original.iloc[0]
    
    # 验证资产总额
    original_total = latest.get('total_assets', 0) or 0
    restructured_total = restructured[restructured['项目'] == '资产总额']['金额'].sum()
    
    tolerance = original_total * 0.001  # 0.1%容差
    
    if abs(original_total - restructured_total) > tolerance:
        print(f"❌ 资产总额不一致:")
        print(f"   原始: {original_total:,.0f}")
        print(f"   重构: {restructured_total:,.0f}")
        return False
    
    print(f"✅ 资产总额一致: {original_total:,.0f}")
    return True

if __name__ == "__main__":
    from prototype_step2_read_data import read_financial_data
    
    balance, _, _ = read_financial_data("000333", years=1)
    
    print("=== 资产负债表重构 ===")
    restructured = restructure_balance_sheet(balance)
    print(restructured.to_string(index=False))
    
    print("\n=== 校验结果 ===")
    validate_restructure(balance, restructured)
```

```python
"""
原型步骤5: 验证Excel输出
"""
import pandas as pd
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
from pathlib import Path
from datetime import date

def create_restructure_excel(
    stock_code: str,
    balance_df: pd.DataFrame,
    income_df: pd.DataFrame,
    cashflow_df: pd.DataFrame,
    output_dir: str = "output/prototype"
) -> Path:
    """生成重构报表Excel文件"""
    
    # 创建输出目录
    output_path = Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)
    
    # 创建工作簿
    wb = Workbook()
    
    # 样式定义
    header_font = Font(bold=True, size=11)
    header_fill = PatternFill(start_color="D9E1F2", end_color="D9E1F2", fill_type="solid")
    border = Border(
        left=Side(style="thin"),
        right=Side(style="thin"),
        top=Side(style="thin"),
        bottom=Side(style="thin")
    )
    
    # 写入资产负债表
    ws1 = wb.active
    ws1.title = "资产资本表"
    
    # 写入表头
    ws1['A1'] = "项目"
    ws1['B1'] = "金额"
    ws1['A1'].font = header_font
    ws1['A1'].fill = header_fill
    ws1['B1'].font = header_font
    ws1['B1'].fill = header_fill
    
    # 写入数据
    for idx, row in balance_df.iterrows():
        ws1.cell(row=idx+2, column=1, value=row['项目'])
        ws1.cell(row=idx+2, column=2, value=row['金额'])
        
        # 应用边框
        ws1.cell(row=idx+2, column=1).border = border
        ws1.cell(row=idx+2, column=2).border = border
        
        # 数字格式
        if isinstance(row['金额'], (int, float)):
            ws1.cell(row=idx+2, column=2).number_format = '#,##0.00'
    
    # 设置列宽
    ws1.column_dimensions['A'].width = 25
    ws1.column_dimensions['B'].width = 20
    
    # 写入利润表（简化）
    ws2 = wb.create_sheet(title="股权价值增加表")
    ws2['A1'] = "项目"
    ws2['B1'] = "TTM"
    ws2['A1'].font = header_font
    ws2['A1'].fill = header_fill
    ws2['B1'].font = header_font
    ws2['B1'].fill = header_fill
    
    # 写入现金流量表（简化）
    ws3 = wb.create_sheet(title="现金流量分析表")
    ws3['A1'] = "项目"
    ws3['B1'] = "TTM"
    ws3['A1'].font = header_font
    ws3['A1'].fill = header_fill
    ws3['B1'].font = header_font
    ws3['B1'].fill = header_fill
    
    # 保存文件
    output_file = output_path / f"{stock_code}_重构报表_{date.today()}.xlsx"
    wb.save(output_file)
    
    return output_file

if __name__ == "__main__":
    # 使用前面的数据
```
